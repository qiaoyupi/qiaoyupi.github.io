Set.prototype.union = function (val) {
  const union = new Set()
  this.forEach((el) => {
    union.add(el)
  })
  val.forEach((elm) => {
    union.add(elm)
  })
  return union
}

new Set([1, 2, 3, 4]).union(new Set([3, 4, 5, 6]))
// 1...6

Set.prototype.intersection = function (val) {
  const intersection = new Set()
  if(this.size > val.size) {
    val.forEach((el) => {
      if (this.has(el)) {
        intersection.add(el)
      }
    })
  } else {
    this.forEach((el) => {
      if (val.has(el)) {
        intersection.add(el)
      }
    })
  }
  return intersection
}

new Set([1, 2, 3, 4]).intersection(new Set([3, 4, 5, 6]))

Set.prototype.difference = function (val) {
  const difference = new Set()
  this.forEach((el) => {
    if (!val.has(el)) {
      difference.add(el)
    }
  })
  return difference
}

new Set([1, 2, 3, 4]).difference(new Set([3, 4, 5, 6]))

Set.prototype.subset = function (val /* B */) {
  if (this.size > val.size) {
    return false
  }

  let f = true
  this.forEach((el) => {
    if (f && !val.has(el)) {
      f = false
    }
  })
  return f
}

new Set([1, 2, 3, 4]).subset(new Set([3, 4, 5, 6]))
// false
new Set([3, 4]).subset(new Set([3, 4, 5, 6]))
